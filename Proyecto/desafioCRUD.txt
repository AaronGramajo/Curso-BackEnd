    i) db.productos.find({price: {$lt: 1000}})
    ii) db.productos.find({$and: [{price: {$gte: 1000}}, {price: {$lte: 3000}}]}) || // si lo queremos ordenado // db.productos.find({$and: [{price: {$gte: 1000}}, {price: {$lte: 3000}}]}).sort({'price': 1})

c) db.productos.updateMany({}, {$set: {'stock': 100}}) // actualizamos el stock a 100 a todos los productos. 
    o si hay que crearlo se haria con : db.productos.updateMany({}, {$set: {'stock': 100}}, {upsert: true})

d) db.productos.updateMany({'price': {$gt: 4000}}, {$set: {'stock': 0}})  // cambia el stock a todos los prodcutos mayores de 4000 a 0.

e) db.productos.deleteMany({'price': {$lt: 1000}}) // borra todos los productos con el precio menos de 1000
